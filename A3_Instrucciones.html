# ------------------------------------- # 🚗 Análisis de autos - Dataset mtcars # ------------------------------------- # 📥 Cargar datos desde internet cat("🔄 Cargando datos...\n") url <- "https://raw.githubusercontent.com/plotly/datasets/master/mtcars.csv" datos <- read.csv(url) cat("✅ ¡Datos cargados exitosamente!\n") cat("📊 Número de autos en el dataset:", nrow(datos), "\n\n") # 📋 Mostrar tabla de datos (primeras 8 filas) cat("📋 TABLA DE DATOS (primeras 8 filas):\n") cat(strrep("=", 50), "\n") print(head(datos, 8)) cat(strrep("=", 50), "\n\n") # 📈 Resumen estadístico de variables clave cat("📈 RESUMEN ESTADÍSTICO:\n") cat("Variables: MPG (millas por galón), HP (caballos de fuerza), WT (peso)\n\n") print(summary(datos[c("mpg", "hp", "wt")])) cat(strrep("=", 50), "\n\n") # 📊 Crear gráfico de dispersión entre Peso y Consumo cat("📊 CREANDO GRÁFICO...\n") plot( datos$wt, datos$mpg, main = "Relación entre Peso y Consumo de Combustible", xlab = "Peso del auto (1000 lbs)", ylab = "Millas por galón (MPG)", pch = 19, # Puntos sólidos col = "blue", # Color azul cex = 1.2 # Tamaño de puntos ) # 📉 Añadir línea de tendencia (regresión lineal) abline(lm(mpg ~ wt, data = datos), col = "red", lwd = 2) cat("🎯 ¡Análisis completado!\n") cat("💡 El gráfico muestra que autos más pesados consumen más combustible.\n")

Presiona Enter después de pegar el código completo

  • 📈 Resumen estadístico de MPG, HP, WT
  • 📊 Gráfico de dispersión azul con línea roja
  • 🎯 Mensaje "¡Análisis completado!"
